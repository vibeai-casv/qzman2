    <script>
        // Real-time WebSocket connection
        const socket = new WebSocket('ws://localhost:8000/ws/qm/');
        
        socket.onmessage = function(e) {
            const data = JSON.parse(e.data);
            console.log('QM Update:', data);
            
            // Handle different message types
            switch(data.type) {
                case 'buzz':
                    handleBuzzer(data.team, data.time);
                    break;
                case 'answer':
                    handleAnswer(data.team, data.answer, data.correct);
                    break;
                case 'timer_update':
                    updateTimer(data.remaining);
                    break;
            }
        };
        
        function handleBuzzer(teamName, responseTime) {
            // Show buzzer notification

                        document.getElementById('buzzerInfo').textContent = 'Wait for QM to open buzzer';
                    }
                    break;
                    
                case 'question_update':
                    document.getElementById('questionText').textContent = data.question;
                    updateOptions(data.options);
                    break;
                    
                case 'answer_feedback':
                    showAnswerFeedback(data);
                    break;
                    
                case 'timer_update':
                    updateTeamTimer(data.remaining);
                    break;
            }

// sw.js - Service Worker for Team PWA
const CACHE_NAME = 'qzman-pwa-v2';
const OFFLINE_URL = '/offline.html';

// Assets to cache on install
const PRECACHE_ASSETS = [
  '/',
  '/styles/main.css',
  '/scripts/app.js',
  '/images/icon-192.png',
  '/images/icon-512.png',
  '/manifest.json',
  OFFLINE_URL
];

// Install event
self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(PRECACHE_ASSETS))
      .then(() => self.skipWaiting())
  );
});

// Activate event
self.addEventListener('activate', event => {
  event.waitUntil(
    caches.keys().then(cacheNames => {
      return Promise.all(
        cacheNames.map(cacheName => {
          if (cacheName !== CACHE_NAME) {
            return caches.delete(cacheName);
          }
        })
      );
    }).then(() => self.clients.claim())
  );
});

// Fetch event with network-first strategy for API, cache-first for assets
self.addEventListener('fetch', event => {
  if (event.request.mode === 'navigate') {
    event.respondWith(
      fetch(event.request)
        .catch(() => caches.match(OFFLINE_URL))
    );
    return;
  }

  event.respondWith(
    caches.match(event.request)
      .then(cachedResponse => {
        if (cachedResponse) {
          return cachedResponse;
        }
        
        return fetch(event.request)
          .then(response => {
            // Don't cache non-GET requests or non-successful responses
            if (event.request.method !== 'GET' || !response.ok) {
              return response;
            }
            
            // Clone the response
            const responseToCache = response.clone();
            
            caches.open(CACHE_NAME)
              .then(cache => {
                cache.put(event.request, responseToCache);
              });
              
            return response;
          })
          .catch(() => {
            // Return offline page for navigation requests
            if (event.request.mode === 'navigate') {
              return caches.match(OFFLINE_URL);
            }
          });
      })
  );
});

// Background sync for offline answers
self.addEventListener('sync', event => {
  if (event.tag === 'sync-answers') {
    event.waitUntil(syncAnswers());
  }
});

async function syncAnswers() {
  const db = await openAnswersDB();
  const answers = await db.getAll('pendingAnswers');
  
  for (const answer of answers) {
    try {
      await fetch('/api/submit-answer', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(answer)
      });
      
      await db.delete('pendingAnswers', answer.id);
    } catch (error) {
      console.log('Sync failed for answer:', answer.id);
    }
  }
}

function openAnswersDB() {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open('qzman-answers', 1);
    
    request.onupgradeneeded = function(event) {
      const db = event.target.result;
      if (!db.objectStoreNames.contains('pendingAnswers')) {
        db.createObjectStore('pendingAnswers', { keyPath: 'id' });
      }
    };
    
    request.onsuccess = function(event) {
      resolve(event.target.result);
    };
    
    request.onerror = function(event) {
      reject(event.target.error);
    };
  });
}

Development Commands:
bash
# Backend (Django)
cd backend
pip install -r requirements.txt
python manage.py migrate
python manage.py createsuperuser
python manage.py runserver

# Frontend (React)
cd frontend
npm install
npm run build
npm start
